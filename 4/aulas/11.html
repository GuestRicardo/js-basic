Correção de segurança para usuarios

Primeiramente precisaremos ocultar os dados de hash e senha, e mostrar somente o desejado, no arquivo:

UserController.js
//index
async index(req, res) {
try {
const users = await User.findAll({ attributes: ['id', 'nome', 'email'] });
return res.json(users);
} catch (erro) {
return res.json(null);
}
};
//show
async show(req, res) {
try {
const user = await User.findByPk(req.params.id);
const {id, nome, email} = user;
return res.json(id, nome, email);
} catch (erro) {
return res.json(null);
}
};
//update
async show(req, res) {
try {
const user = await User.findByPk(req.userId);
if (!user) {
return res.status(400).json({
errors: ['USUÁRIO não existe.'],
});
}
const novosDados = await user.update(req.body); return res.json(novosDados);
} catch (erro) {
res.status(400).json({
erros: erro.errors.map((err) => err.mensage),
});
}
};
