Escreva 2 medidas para saber se a imagem esta no modo paisagem ou modo retrato



Criando esse exercicio usando function.
function maior(n1, n2){
if (n1 > n2){
return `O ${n1} é maior que o ${n2}`
}
return `O ${n2} é maior que o ${n1}`
}
console.log(maior(25,5));
