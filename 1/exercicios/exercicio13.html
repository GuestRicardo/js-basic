numeno é divisivel por 5 = BUZZ
numeno é divisivel por 3 e 5 = FIZZBUZZ
numeno NÂO é divisivel po 3 e 5= retorne o proprio numero

Resolução
function fizzBuzz(numero){
if(typeOf(numero) !== 'numero') return NaN;
if (numero % 3 === 0 && numero % 5 === 0) return 'FizzBuzz';
if (numero % 3 === 0) return 'Fizz';
if (numero % 5 === 0) return 'Buzz';
return numero
}
