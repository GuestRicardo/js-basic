Switch / Case

O switch é usado para comparar a igualdade de valores com uma expressão (em geral uma variável é usada).
funciona muito parecido com if e else, mais ele é recomendado para situações especificas, exemplo um menu.

exemplo:

neste exemplo sera feita uma verificação do da data verificada.

const data = new Date("1987-03-23 00:51:05");
let diaSemana = data.getDay();
let diaSemanaTexto;

switch (diaSemana) {
case0:
diaSemana = 'Domingo';
break;
case1:
diaSemana = 'Segunda-Feira';
break;
case2:
diaSemana = 'Terça-Feira';
break;
case3:
diaSemana = 'Quarta-Feira';
break;
case4:
diaSemana = 'Quinta-Feira';
break;
case5:
diaSemana = 'Sexta-Feira';
break;
case6:
diaSemana = 'Sabado';
break;
default:
diaSemanaTexto = 'informação invalida';
break;
}
console.log(diaSemana, diaSemanaTexto);

Mesmo exemplo acima usando uma função

function getDiaSemanaTexto(diaSemana){
let diaSemanaTexto;
switch (diaSemana) {
case0:
diaSemana = 'Domingo';
return diaSemanaTexto;
case1:
diaSemana = 'Segunda-Feira';
return diaSemanaTexto;;
case2:
diaSemana = 'Terça-Feira';
return diaSemanaTexto;;
case3:
diaSemana = 'Quarta-Feira';
return diaSemanaTexto;
case4:
diaSemana = 'Quinta-Feira';
return diaSemanaTexto;
case5:
diaSemana = 'Sexta-Feira';
return diaSemanaTexto;
case6:
diaSemana = 'Sabado';
return diaSemanaTexto;
default:
diaSemanaTexto = 'informação invalida';
return diaSemanaTexto;
}
}
const diaSemanaTexto = getDiaSemanaTexto(diaSemana);
