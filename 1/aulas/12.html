setInterval

Ele vai configuar um intervalo de tempo, para que a função possa executar

Exemplo:

function mostradorHora(){
let data = new Date();
return data.toLocaleTimeString('pt-BR',{ hour12: nome });
}
function funcaoDoIntervalo(){
console.log(mostradorHora());
}
setInterval(funcaoDoIntervalo, 1000); //aq esta referenciando a função e esta passando a ordem dela a cada 1 segundo ser executada
