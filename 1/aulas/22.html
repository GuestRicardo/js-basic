Heranças com Classes

É uma forma elegante e simples,

class DispositivoEletronico{
constructor(nome){
this.nome = nome;
this.ligado = false;
}
//metodos
ligar(){
if(this.ligado){
console.log( this.nome + 'já ligado');
return;
}
//estado
this.ligado = true;
}
desligar(){
if(!this.ligado){
console.log( this.nome + 'já desligado');
return;
}
//estado
this.ligado = false;
}
}
//herança
class Tv extends DispositivoEletronico{
}
const dispositivo1 = new Tv('TV samptoshiba');
dispositivo1.ligar();
dispositivo1.desligar();
console.log(dispositivo1);
